rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Funciones auxiliares
    function isAuthenticated() {
      return request.auth != null;
    }
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // --- USUARIOS ---
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && request.resource.data.id == request.auth.uid;
      allow update, delete: if isOwner(userId);
    }
    
    // --- PROFESORES ---
    match /teachers/{teacherId} {
      // Todo el mundo puede ver la lista de profes
      allow read: if true;
      // Solo el profe due√±o puede crear/editar su ficha
      allow create: if isAuthenticated() && request.resource.data.id == request.auth.uid;
      allow update: if isOwner(teacherId);
    }

    // --- SOLICITUDES / CLASES ---
    match /requests/{requestId} {
      allow create: if isAuthenticated();
      allow read, update: if isAuthenticated() && (
        resource.data.studentId == request.auth.uid || 
        resource.data.teacherId == request.auth.uid
      );
    }
    
    // --- CHAT y MENSAJES ---
    match /messages/{messageId} {
      allow create: if isAuthenticated();
      // Solo los participantes ven los mensajes
      allow read: if isAuthenticated() && (
        resource.data.studentId == request.auth.uid || 
        resource.data.teacherId == request.auth.uid
      );
    }
  }
}
